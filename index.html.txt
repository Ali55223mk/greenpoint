<script>
    const FB_URL = "https://greenpoint-1-default-rtdb.firebaseio.com/";
    let currentUserEmailKey = "";
    let selectedGift = { name: "", price: 0 };

    async function login() {
        const n = document.getElementById('uN').value, e = document.getElementById('uE').value.toLowerCase().trim();
        if(!n || !e) return alert("البيانات ناقصة!");
        const emailKey = e.replace(/\./g, '_');
        const res = await fetch(`${FB_URL}users/${emailKey}.json`);
        let user = await res.json();
        if(!user) {
            user = {name: n, email: e, pts: 0};
            await fetch(`${FB_URL}users/${emailKey}.json`, { method: 'PUT', body: JSON.stringify(user) });
        }
        localStorage.setItem("GP_SESSION", emailKey);
        location.reload(); 
    }

    function checkSession() {
        const session = localStorage.getItem("GP_SESSION");
        if(session) {
            currentUserEmailKey = session;
            document.getElementById('authBox').classList.add('hidden');
            document.getElementById('appBox').classList.remove('hidden');
            startSync();
        }
    }

    function startSync() { fetchData(); setInterval(fetchData, 4000); }

    async function fetchData() {
        try {
            const res = await fetch(`${FB_URL}.json`);
            const data = await res.json();
            if(data) renderUI(data);
        } catch(e) {}
    }

    function renderUI(data) {
        const me = data.users[currentUserEmailKey];
        if(me) {
            document.getElementById('meName').innerText = me.name;
            document.getElementById('mePts').innerText = me.pts;
        }

        document.getElementById('listT').innerHTML = Object.keys(data.tasks || {}).map(id => `
            <div class="card">
                <span><b>${data.tasks[id].t}</b><br><small style="color:var(--primary)">+${data.tasks[id].p} نقطة</small></span>
                <button class="btn-main" onclick="sendReq('${data.tasks[id].t}', ${data.tasks[id].p})">إنجاز</button>
            </div>`).join('') || '<p style="text-align:center">لا مهام</p>';

        document.getElementById('listS').innerHTML = Object.keys(data.gifts || {}).map(id => `
            <div class="card">
                <span><b>${data.gifts[id].t}</b><br><small style="color:var(--gold)">السعر: ${data.gifts[id].p}</small></span>
                <button class="btn-buy" onclick="openOrderModal('${data.gifts[id].t}', ${data.gifts[id].p})">طلب 💰</button>
            </div>`).join('') || '<p style="text-align:center">المتجر فارغ</p>';

        const usersArr = Object.values(data.users || {}).sort((a,b) => b.pts - a.pts);
        document.getElementById('listR').innerHTML = usersArr.map((u, i) => `
            <div class="card" style="${u.email.replace(/\./g, '_') === currentUserEmailKey ? 'border: 1px solid var(--primary); background:rgba(0,255,136,0.05)' : ''}">
                <span><b>${i+1}. ${u.name}</b></span><b style="color:var(--primary)">${u.pts}</b>
            </div>`).join('');
    }

    function openOrderModal(giftName, price) {
        const currentPts = parseInt(document.getElementById('mePts').innerText);
        if(currentPts < price) return alert("نقاطك لا تكفي!");
        selectedGift = { name: giftName, price: price };
        document.getElementById('giftInfo').innerText = `أنت تطلب: ${giftName} بسعر ${price} نقطة`;
        document.getElementById('orderOverlay').style.display = 'flex';
    }

    function closeOrder() { document.getElementById('orderOverlay').style.display = 'none'; }

    async function submitOrder() {
        const phone = document.getElementById('orderPhone').value;
        const addr = document.getElementById('orderAddress').value;
        if(!phone || !addr) return alert("يرجى إكمال البيانات!");
        const orderData = {
            uN: document.getElementById('meName').innerText,
            uE: currentUserEmailKey,
            t: `🎁 جائزة: ${selectedGift.name}`,
            p: selectedGift.price,
            proof: `📞 الهاتف: ${phone} \n📍 العنوان: ${addr}`
        };
        await fetch(`${FB_URL}reqs/${Date.now()}.json`, { method: 'PUT', body: JSON.stringify(orderData) });
        alert("تم إرسال طلبك للإدارة بنجاح! 🚚");
        closeOrder();
    }

    async function sendReq(t, p) {
        const pr = prompt("رابط إثبات المهمة:");
        if(pr) {
            const reqData = { uN: document.getElementById('meName').innerText, uE: currentUserEmailKey, t: t, p: p, proof: pr };
            await fetch(`${FB_URL}reqs/${Date.now()}.json`, { method: 'PUT', body: JSON.stringify(reqData) });
            alert("تم الإرسال!");
        }
    }

    function nav(p, el) {
        ['H','S','R'].forEach(x => document.getElementById('pg-'+x).classList.add('hidden'));
        document.querySelectorAll('.tab').forEach(t => t.classList.remove('active'));
        document.getElementById('pg-'+p).classList.remove('hidden'); el.classList.add('active');
    }

    function logout() { if(confirm("خروج؟")) { localStorage.removeItem("GP_SESSION"); location.reload(); } }

    checkSession();
</script>
